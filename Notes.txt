    def extract():
        # Create an empty data frame to hold extracted data
        extracted_data = pd.DataFrame(columns=['name','height','weight'])

        #process all csv files
        for csvfile in glob.glob('*.csv'):
            extracted_data = extracted_data.append(extract_from_csv(csvfile), ignore_index=True)

        for jsonfile in glob.glob('*.json'):
            extracted_data=extracted_data.append(extract_from_json(jsonfile),ignore_index=True)

        return extracted_data

    def transform(data):
        #Convert height from inches to meters
        data['height']=round(data.height*0.0254,2)

        #Convert pounds to kilograms
        data['weight']=round(data.weight*0.45359237,2)
        return data

Loging progress

    from datetime import datetime
    def  log(message):
        timestamp_format = '%Y-%h-%d-%H:%M:%S'
        now = datetime.now()
        timestamp = now.strftime(timestamp_format)
        with open('logfile.txt','a') as f:
            f.write(timestamp+',' + message + '\n')

Complete process
    log('ETL Job started')

    log('Extract phase Started')
    extracted_data = extract()
    log('Extract phase ended')

    log('Transform Job Started")
    transformed_data = transform(extracted_data)
    log('Transform Job Ended')

    log('Load Job Started')
    load(targetfile,transformed_data)
    log('Load Job Ended')

    log('ETL Job ended')
	

WEB SCRAPPING
=============
Requests, httplib, urllib work with the HTTP protocol.

Requests sends HTTP/1.1 requests.

    import requests
	
	url = 'http://www.ibm.com/'
	r=requests.get(url)
	if (r.status_code==200):
	    print("server returned OK")
	else:
		print(f"Server returned error code {r.status_code}")

	# request headers come as .request.headers attributes,
	# which is a dictionary accessible by item name
	r.request.headers['date'] # gets the string 'Thu, 19 Nov 2020 15:21:47 GMT'
	r.request.headers['Content-Type'] # gets the string 'text/html; charset=UTF-8'
	
	r.body # accesses the response body (None for a GET request)

	r.encoding # gets 'UTF-8'
	r.text # gets the HTML in the body

You can use a URL with parameters to alter the query results.
You can train against http://httpbin.org with /get&<parameters>
You can also send data to a service with a POST request with data in the body.
In a POST request you pass the data in a payload dictionary.

SQLite3 database
================
Pandas can interface directly with a SQLite3 database
import pandas as pd
import sqlite3

sql_connection=sqlite3.connect('database.db')
#create a database table from a dataframe
df.to_sql(table_name,sql_connection,if_exists='fail',index=False)

#overwrite a database table
df.to_sql(table_name,sql_connection,if_exists='replace',index=False)

#append to a database table
df.to_sql(table_name,sql_connection,if_exists='append',index=False)

#read from a database table
df = pandas.read_sql(query_statement, sql_connection)
